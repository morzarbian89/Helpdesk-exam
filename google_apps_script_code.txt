function doGet(e) {
  setup();
  const action = e.parameter.action || '';
  const callback = e.parameter.callback || 'callback';
  if (action === 'questions') {
    const data = getQuestions_();
    return ContentService.createTextOutput(callback + '(' + JSON.stringify(data) + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput('OK');
}

function doPost(e) {
  setup();
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName(RESULTS_SHEET);
  const data = JSON.parse(e.postData.contents || '{}');
  sh.appendRow([
    new Date(), data.candidateName || '', data.phone || '', data.email || '', data.position || '',
    data.score || 0, data.correct || 0, data.total || 0, data.durationSeconds || 0,
    data.endedByTime || '', JSON.stringify(data.categoryScores || {}), JSON.stringify(data.answers || [])
  ]);
  return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}

function setup() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let q = ss.getSheetByName(QUESTIONS_SHEET) || ss.insertSheet(QUESTIONS_SHEET);
  let r = ss.getSheetByName(RESULTS_SHEET) || ss.insertSheet(RESULTS_SHEET);
  if (q.getLastRow() === 0) {
    q.appendRow(['ID','Category','Difficulty','Question','Answer1','Answer2','Answer3','Answer4','Correct','Active']);
    q.getRange(2,1,DEFAULT_QUESTIONS.length,DEFAULT_QUESTIONS[0].length).setValues(DEFAULT_QUESTIONS);
  }
  if (r.getLastRow() === 0) {
    r.appendRow(['Timestamp','Candidate Name','Phone','Email','Position','Score','Correct','Total','Duration Seconds','Ended By Time','Category Scores','Full Answers']);
  }
}

function getQuestions_() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName(QUESTIONS_SHEET);
  const values = sh.getDataRange().getValues();
  const out = [];
  for (let i = 1; i < values.length; i++) {
    const row = values[i];
    if (String(row[9]).toLowerCase() === '0' || String(row[9]).toLowerCase() === 'false') continue;
    out.push({
      id: row[0], category: row[1], difficulty: row[2], text: row[3],
      answers: [row[4], row[5], row[6], row[7]], correct: Number(row[8]) - 1
    });
  }
  return out;
}
